def applyStructureToImage(src, sType):
    if sType == 'circle':
        circleStruct = cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (12,12))
        srcStructCircle = cv2.dilate(src,circleStruct ,iterations=1)
        return srcStructCircle
    elif sType == 'square':
        squareStrcut = cv2.getStructuringElement(cv2.MORPH_RECT, (12, 12))
        srcStructSquare = cv2.dilate(src, squareStrcut, iterations=1)
        return srcStructSquare

def applyEdgeDetection(src, kernel_shape = (5,5)):
    kernel = np.ones(kernel_shape,np.uint8)
    src_edged = cv2.morphologyEx(src, cv2.MORPH_GRADIENT, kernel)
    return src_edged

def applyErodeStructureToImg(src, sType):
    if sType == 'circle':
        circleStruct = cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (12,12))
        srcStuctCircle = cv2.erode(src, circleStruct, iterations = 1)
        return srcStuctCircle
    elif sType == 'square':
        squareStruct = cv2.getStructuringElement(cv2.MORPH_RECT, (12,12))
        srcStructSquare = cv2.erode(src, squareStruct, iterations = 1)
        return srcStructSquare

def applyThresholdAndErosion(src):
    _, thresh = cv2.threshold(src, 88, 252, cv2.THRESH_BINARY_INV)
    circleStruct = cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (12,12))
    src_erode = cv2.erode(thresh, circleStruct, iterations=1)
    return src_erode
